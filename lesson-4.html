Применение циклических конструкций

  • Формирование знаний о необходимости применения циклов, их обозначении на рабочем языке курса.
  • Знакомство с понятиями «вложенные циклы», «прерывание цикла», а также с примерами использования циклических конструкций.
  • Учащиеся получат навыки пошагового выполнения цикла в режиме отладки.

Формирование знаний о необходимости применения циклов

Цикл со счётчиком (или цикл для)

for (int i = 0; i < n; i++) {
	console.log(i)
}

Цикл с постусловием

let a = 0
while (a < 5) {
	console.log(a)
	a++
}

Цикл с предусловием

let a = 0
do {
	console.log(a)
	a++
} while(a < 5)

Цикл с выходом из середины

let a = 0
do {
	console.log(a)
	a++
	if (a == 3) {
		break
	}
} while(a < 5)

Безусловные циклы

for (;;) { ... }
do { ... } while (true)
while (1) { ... }

Совместный цикл

let array = ['cat', 'dog', 'mouse', 'horse']
for (animal of array) { 
	console.log(animal)
}

Вложенный цикл

1 2 3
1 X O O
2 O X X
3 X O X 
let result = []
const data = [['X', 'O', 'O'], ['O', 'X', 'X'], ['X', 'O', 'X']]

for (let i = 0; i < data.length; i++) {
	for (let j = 0; j < data[i].length; j++) {
		if (data[i][j] == 'O') {
			result.push({
				row: i + 1,
				column: j + 1
			})
		}
	}
}
console.log(result)

Бонус: рекурсия

Pекурсия — вызов функции (процедуры) из неё же самой

sumTo(1) = 1
sumTo(2) = 2 + 1 = 3
sumTo(3) = 3 + 2 + 1 = 6
sumTo(4) = 4 + 3 + 2 + 1 = 10
...
sumTo(100) = 100 + 99 + ... + 2 + 1 = 5050

Рекурсия

function sumTo(n) {
	if (n == 1) {
		return 1
	}
	return n + sumTo(n - 1)
}
  
console.log( sumTo(100) )

Цикл

function sumTo(n) {
	let sum = 0
	for (let i = 1; i <= n; i++) {
		sum += i
	}
	return sum
}
  
alert( sumTo(100) )
